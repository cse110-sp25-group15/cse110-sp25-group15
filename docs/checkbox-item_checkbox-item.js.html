checkbox-item/checkbox-item.js

import templateHTML from './checkbox-item.html?raw';
import templateCSS from './checkbox-item.css?raw';

/**
 * Custom checkbox component that can be toggled on and off.
 * @class
 * @extends HTMLElement
 */
class CheckItem extends HTMLElement {
  /**
   * Attributes to observe for changes.
   * @static
   * @returns {string[]} Array of attribute names to observe.
   */
  static get observedAttributes() {
    return ['checked'];
  }

  /**
   * Creates an instance of CheckItem and attaches a shadow DOM.
   * @constructor
   */
  constructor() {
    super();
    this.attachShadow({ mode: 'open' });
  }

  /**
   * Lifecycle callback when the element is added to the DOM.
   * Sets up the component template, event listeners, and initial state.
   */
  connectedCallback() {
    const template = document.createElement('template');
    template.innerHTML = `
      <style>${templateCSS}</style>
      ${templateHTML}
    `;
    this.shadowRoot.appendChild(template.content.cloneNode(true));

    this.checkbox = this.shadowRoot.querySelector('.checkbox');

    if (this.checkbox) {
      this.checkbox.addEventListener('click', () => {
        this.toggleAttribute('checked');
      });
    }

    this.updateCheckedStyle();
  }

  /**
   * Lifecycle callback that is called when an observed attribute changes.
   * @param {string} name - The name of the attribute that changed.
   */
  attributeChangedCallback(name) {
    if (name === 'checked') {
      this.updateCheckedStyle();
    }
  }

  /**
   * Updates the visual state of the checkbox based on the 'checked' attribute.
   * Toggles the 'checked' class on the checkbox element.
   * @private
   */
  updateCheckedStyle() {
    if (!this.checkbox) { return; }
    this.checkbox.classList.toggle('checked', this.hasAttribute('checked'));
  }
}

customElements.define('checkbox-item', CheckItem);
